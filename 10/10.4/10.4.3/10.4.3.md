# 10.4.3. Módulo 3 / Integrante 3

# Funciones ⨍

## Se crean funciones para encapsular la lógica de negocio, validaciones y procesos transaccionales del módulo de almacén.

### 1. Hallar el Stock Disponible Real de un Producto

Esta función es vital y puede ser consumida por otros módulos (Ventas, Transporte) para validar la disponibilidad. Suma el stock de todas las ubicaciones de 'VENTA' de un producto y resta el stock comprometido.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".fn_get_stock_disponible_producto(
    p_cod_producto INTEGER
)
RETURNS BIGINT AS $$
DECLARE
    v_stock_disponible BIGINT;
BEGIN
    SET search_path TO "FERRETERIA";

    SELECT COALESCE(SUM(i.stock_fisico - i.stock_comprometido), 0)
    INTO v_stock_disponible
    FROM inventario i
    JOIN ubicacion u ON i.cod_ubicacion = u.cod_ubicacion
    WHERE i.cod_producto = p_cod_producto
      AND u.tipo_ubicacion = 'VENTA'; -- Clave: Solo contar stock vendible

    RETURN v_stock_disponible;
END;
$$ LANGUAGE plpgsql;

-- Probar con un ID de producto que exista
SELECT "FERRETERIA".fn_get_stock_disponible_producto(45); -- (ej. Cemento)

```

### 2. Generar Cupos de Almacén (Proceso BATCH)

Esta función es el motor del **R-307 (Gestión de Capacidad)**. Lee las reglas de capacidad (`capacidad_turno`) y genera (inserta) las filas de `cupo_disponible` para un rango de fechas. Es similar al `sp_generar_reporte` del Módulo 1.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".sp_generar_cupos_almacen(
    p_fecha_inicio DATE,
    p_dias_a_generar INT
)
RETURNS INTEGER AS $$ -- Devuelve el total de cupos creados
DECLARE
    v_total_cupos_creados INT := 0;
    v_fecha_actual DATE;
    v_dia_semana INT;
    v_regla_capacidad RECORD;
BEGIN
    SET search_path TO "FERRETERIA";

    RAISE NOTICE 'Iniciando generación de cupos desde % por % días...', p_fecha_inicio, p_dias_a_generar;

    -- Bucle principal: 1 iteración por cada día
    FOR i IN 0..(p_dias_a_generar - 1) LOOP
        v_fecha_actual := p_fecha_inicio + i;
        v_dia_semana := EXTRACT(ISODOW FROM v_fecha_actual); -- 1=Lunes, 7=Domingo

        -- Bucle anidado: Por cada día, buscar todas las reglas de turno/instalación
        FOR v_regla_capacidad IN (
            SELECT * FROM "FERRETERIA".capacidad_turno
            WHERE dia_semana = v_dia_semana
        ) LOOP
            -- Bucle interno: Crear N cupos según la capacidad total
            FOR j IN 1..v_regla_capacidad.capacidad_total LOOP
                INSERT INTO "FERRETERIA".cupo_disponible (
                    fecha_cupo,
                    estado,
                    cod_turno,
                    cod_instalacion
                ) VALUES (
                    v_fecha_actual,
                    'Disponible',
                    v_regla_capacidad.cod_turno,
                    v_regla_capacidad.cod_instalacion
                );
                v_total_cupos_creados := v_total_cupos_creados + 1;
            END LOOP; -- fin bucle j
        END LOOP; -- fin bucle v_regla_capacidad
    END LOOP; -- fin bucle i

    RAISE NOTICE 'Generación completada. Total de cupos creados: %', v_total_cupos_creados;
    RETURN v_total_cupos_creados;
END;
$$ LANGUAGE plpgsql;

-- Genera los cupos para los próximos 30 días
SELECT "FERRETERIA".sp_generar_cupos_almacen(NOW()::DATE, 30);

```

### 3. Procesar Salida de Picking (Transaccional)

Esta función transaccional encapsula la lógica de **Postcondición del R-306 (Picking)**. Cuando el operador "Finaliza Picking" de una reserva de despacho, esta función:

1. Encuentra todos los productos/cantidades de ese despacho.
2. Crea los `MOVIMIENTO` de tipo 'SALIDA'.
3. Actualiza (reduce) el `stock_fisico` y `stock_comprometido` en `INVENTARIO`.
4. Marca la reserva de almacén como 'Completado'.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".sp_procesar_picking_salida(
    p_cod_reserva_almacen VARCHAR(50)
)
RETURNS BOOLEAN AS $$
DECLARE
    v_cod_despacho INT;
    v_detalle_picking RECORD;
    v_cod_inventario INT;
BEGIN
    SET search_path TO "FERRETERIA";

    -- 1. Obtener el despacho asociado a la reserva de picking
    SELECT cod_despacho INTO v_cod_despacho
    FROM reserva_almacen
    WHERE cod_reserva = p_cod_reserva_almacen AND tipo_reserva = 'Despacho';

    IF v_cod_despacho IS NULL THEN
        RAISE EXCEPTION 'Reserva % no encontrada o no es de tipo Despacho.', p_cod_reserva_almacen;
    END IF;

    -- 2. Bucle por cada producto/cantidad del despacho
    -- (Consulta compleja de R-306 para obtener la lista de picking)
    FOR v_detalle_picking IN (
        SELECT
            dptr.cod_producto,
            SUM(dptr.cantidad_detalle) AS cantidad_total
        FROM "FERRETERIA".ASIGNACION_PEDIDO_DESPACHO apd
        JOIN "FERRETERIA".PEDIDO_TRANSPORTE pt ON apd.cod_pedido_transporte = pt.cod_pedido_transporte
        JOIN "FERRETERIA".DETALLE_PEDIDO_TR dptr ON pt.cod_pedido_transporte = dptr.cod_pedido_transporte
        WHERE apd.cod_despacho = v_cod_despacho
        GROUP BY dptr.cod_producto
    ) LOOP
        -- 3. Encontrar el inventario de VENTA para este producto
        -- (Asume una sola ubicación de venta por simplicidad, o se debe agregar lógica de ubicación)
        SELECT i.cod_inventario INTO v_cod_inventario
        FROM inventario i
        JOIN ubicacion u ON i.cod_ubicacion = u.cod_ubicacion
        WHERE i.cod_producto = v_detalle_picking.cod_producto
          AND u.tipo_ubicacion = 'VENTA'
        LIMIT 1; -- Asunción clave: el picking sale de la ubicación de 'VENTA'

        IF v_cod_inventario IS NULL THEN
            RAISE EXCEPTION 'No se encontró inventario de VENTA para el producto %', v_detalle_picking.cod_producto;
        END IF;

        -- 4. Crear el Movimiento de Salida
        INSERT INTO "FERRETERIA".movimiento (
            tipo_movimiento,
            cantidad,
            cod_inventario
            -- Aquí se podrían enlazar los cod_venta o cod_detalle_pedido_tr si el modelo lo requiere
        ) VALUES (
            'SALIDA',
            v_detalle_picking.cantidad_total,
            v_cod_inventario
        );

        -- 5. Actualizar el Inventario (Trigger o manual)
        -- Esto cumple la Postcondición R-306, paso 3
        UPDATE "FERRETERIA".inventario
        SET
            stock_fisico = stock_fisico - v_detalle_picking.cantidad_total,
            stock_comprometido = stock_comprometido - v_detalle_picking.cantidad_total
        WHERE
            cod_inventario = v_cod_inventario;

    END LOOP;

    -- 6. Marcar la reserva como completada
    UPDATE "FERRETERIA".reserva_almacen
    SET estado = 'Completado'
    WHERE cod_reserva = p_cod_reserva_almacen;

    RAISE NOTICE 'Picking de reserva % (Despacho %) procesado exitosamente.', p_cod_reserva_almacen, v_cod_despacho;
    RETURN TRUE;

EXCEPTION WHEN OTHERS THEN
    RAISE NOTICE 'Error procesando picking: %', SQLERRM;
    RETURN FALSE;
END;
$$ LANGUAGE plpgsql;

-- Probar con el ID de una reserva de Picking que esté 'En Proceso'
SELECT "FERRETERIA".sp_procesar_picking_salida('T-007'); -- (ID de reserva de picking/despacho)

```
