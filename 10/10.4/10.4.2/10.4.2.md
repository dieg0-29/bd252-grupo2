# 10.4.2. Módulo 2 / Integrante 2

# Funciones ⨍

## Se crean funciones para encapsular la lógica de negocio y las consultas repetitivas del módulo.

### Hallar el número de despachos completados por un chofer

Esta función es útil para reportes de productividad. Cuenta cuántos despachos ha completado un operador específico.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".fn_contar_despachos_completados_chofer(
    p_cod_usuario INTEGER
)
RETURNS BIGINT AS
$$
DECLARE
    v_total_despachos BIGINT;
BEGIN
    SELECT
        COUNT(d.cod_despacho) INTO v_total_despachos
    FROM
        "FERRETERIA".DESPACHO d
    JOIN
        "FERRETERIA".ESTADO_DESPACHO ed ON d.cod_estado_despacho = ed.cod_estado_despacho
    WHERE
        d.cod_chofer = p_cod_usuario
        AND ed.descp_estado_despacho = 'Completado';

    RETURN v_total_despachos;
END;
$$ LANGUAGE plpgsql;

-- Pasa el ID de un chofer que exista en tu BD de prueba
SELECT * FROM "FERRETERIA".fn_contar_despachos_completados_chofer(1);
```

### Verificar disponibilidad de un vehículo para una fecha

Esta función es crucial para el flujo primario R-209 (Programar Despacho). Antes de asignar un vehículo, esta función valida que esté "Operativo" y que no tenga ya un despacho "Programado" o "En Ruta" para esa misma fecha.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".fn_verificar_disponibilidad_vehiculo(
    p_cod_vehiculo INTEGER,
    p_fecha_despacho DATE
)
RETURNS BOOLEAN AS
$$
DECLARE
    v_operativo BOOLEAN;
    v_conteo_despachos INTEGER;
BEGIN
    -- 1. Verificar que el vehículo esté 'Operativo'
    SELECT EXISTS (
        SELECT 1
        FROM "FERRETERIA".VEHICULO v
        JOIN "FERRETERIA".ESTADO_VEHICULO ev ON v.cod_estado_vehiculo = ev.cod_estado_vehiculo
        WHERE v.cod_vehiculo = p_cod_vehiculo
          AND ev.descp_estado_vehiculo = 'Operativo'
    ) INTO v_operativo;

    IF NOT v_operativo THEN
        RETURN FALSE; -- No está operativo
    END IF;

    -- 2. Verificar que no tenga despachos en esa fecha
    SELECT COUNT(d.cod_despacho)
    INTO v_conteo_despachos
    FROM "FERRETERIA".DESPACHO d
    JOIN "FERRETERIA".ESTADO_DESPACHO ed ON d.cod_estado_despacho = ed.cod_estado_despacho
    WHERE d.cod_vehiculo = p_cod_vehiculo
      AND d.fecha_despacho = p_fecha_despacho
      AND ed.descp_estado_despacho IN ('Programado', 'En Ruta', 'Picking');

    IF v_conteo_despachos > 0 THEN
        RETURN FALSE; -- Ya está asignado ese día
    END IF;

    RETURN TRUE; -- Está disponible
END;
$$ LANGUAGE plpgsql;

-- Pasa el ID de un vehículo y una fecha para probar
SELECT * FROM "FERRETERIA".fn_verificar_disponibilidad_vehiculo(1, '2025-11-10');
```

### Procedimiento para Cancelar un Despacho Programado (Transaccional)

Esta función encapsula la lógica compleja del R-212 (Eliminar Despacho). Es una transacción que debe desvincular artículos, eliminar visitas y ayudantes, y finalmente eliminar el despacho. Devuelve `TRUE` si fue exitoso o `FALSE` si falló.

```
CREATE OR REPLACE FUNCTION "FERRETERIA".sp_cancelar_despacho_programado(
    p_cod_despacho INTEGER
)
RETURNS BOOLEAN AS $$
DECLARE
    v_cod_estado_actual INTEGER;
    v_cod_estado_programado INTEGER;
    v_cod_estado_pendiente INTEGER;
BEGIN
    -- 1. Obtener IDs de estados clave
    SELECT cod_estado_despacho FROM "FERRETERIA".ESTADO_DESPACHO WHERE descp_estado_despacho = 'Programado' INTO v_cod_estado_programado;
    SELECT cod_estado_detalle_pedido FROM "FERRETERIA".ESTADO_DETALLE_PEDIDO WHERE descp_estado_detalle_pedido = 'Pendiente' INTO v_cod_estado_pendiente;

    -- 2. Verificar que el despacho esté "Programado"
    SELECT cod_estado_despacho
    INTO v_cod_estado_actual
    FROM "FERRETERIA".DESPACHO
    WHERE cod_despacho = p_cod_despacho;

    IF v_cod_estado_actual IS NULL THEN
        RAISE NOTICE 'Despacho % no existe.', p_cod_despacho;
        RETURN FALSE;
    END IF;

    IF v_cod_estado_actual != v_cod_estado_programado THEN
        RAISE NOTICE 'Solo se pueden cancelar despachos en estado "Programado".';
        RETURN FALSE;
    END IF;

    -- 3. Iniciar Transacción (implícita en funciones PL/pgSQL)
    BEGIN
        -- 4. Devolver artículos a 'Pendiente' y desvincularlos de la visita
        UPDATE "FERRETERIA".DETALLE_PEDIDO_TR
        SET
            cod_visita = NULL,
            cod_estado_detalle_pedido = v_cod_estado_pendiente
        WHERE
            cod_visita IN (SELECT cod_visita FROM "FERRETERIA".VISITA_PROGRAMADA WHERE cod_despacho = p_cod_despacho);

        -- 5. Eliminar Asignación de Ayudantes
        DELETE FROM "FERRETERIA".ASIGNACION_AYUDANTE
        WHERE cod_despacho = p_cod_despacho;

        -- 6. Eliminar las Visitas (Paradas) asociadas
        DELETE FROM "FERRETERIA".VISITA_PROGRAMADA
        WHERE cod_despacho = p_cod_despacho;

        -- 7. Eliminar la cabecera del Despacho
        DELETE FROM "FERRETERIA".DESPACHO
        WHERE cod_despacho = p_cod_despacho;

        RAISE NOTICE 'Despacho % cancelado exitosamente.', p_cod_despacho;
        RETURN TRUE;

    EXCEPTION WHEN OTHERS THEN
        RAISE NOTICE 'Error al cancelar despacho: %', SQLERRM;
        RETURN FALSE;
    END;
END;
$$ LANGUAGE plpgsql;

-- Probar con un ID de despacho que esté "Programado"
SELECT * FROM "FERRETERIA".sp_cancelar_despacho_programado(1);
```
