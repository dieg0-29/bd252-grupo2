# 10.4.1. Módulo 1 / Integrante 1

# Funciones ⨍

### Hallar el numero de canjes realizados

```sql
CREATE OR REPLACE FUNCTION MODULO_CLIENTES.contar_canjes_por_maestro(p_cod_maestro INTEGER)
RETURNS BIGINT AS
$$
DECLARE
    v_total_canjes BIGINT;
BEGIN
    SELECT
        COUNT(cod_canje) INTO v_total_canjes
    FROM
        MODULO_CLIENTES.CANJE
    WHERE
        cod_maestro = p_cod_maestro;

    RETURN v_total_canjes;
END;
$$ LANGUAGE plpgsql;

SELECT * FROM MODULO_CLIENTES.contar_canjes_por_maestro(1);
  ```

### Hallar el numero de ventas referidas

```sql
CREATE OR REPLACE FUNCTION MODULO_CLIENTES.contar_referidos_por_maestro(p_cod_maestro INTEGER)
RETURNS BIGINT AS
$$
DECLARE
    v_total_referidos BIGINT;
BEGIN
    SELECT
        COUNT(COD_MAESTRO) INTO v_total_referidos
    FROM
        MODULO_CLIENTES.VENTA
    WHERE
        cod_maestro = p_cod_maestro;

    RETURN v_total_referidos;
END;
$$ LANGUAGE plpgsql;

SELECT * FROM MODULO_CLIENTES.contar_referidos_por_maestro(1);
  ```
